#include #include #include void displayMenu(); double getNumberInput(); char getOperatorInput(); double calculate(double num1, double num2, char operator); int main() { double num1, num2, result; char operator; char choice; do { displayMenu(); num1 = getNumberInput(); operator = getOperatorInput(); num2 = getNumberInput(); result = calculate(num1, num2, operator); if (result != -99999.0) { printf("Result: %.2lf\n", result); } printf("\nDo you want to perform another calculation? (y/n): "); scanf(" %c", &choice); } while (choice == 'y' || choice == 'Y'); printf("Exiting calculator. Goodbye!\n"); return 0; } void displayMenu() { printf("\n--- Simple C Calculator ---\n"); printf("Enter two numbers and an operator (+, -, *, /).\n"); printf("---------------------------\n"); } double getNumberInput() { double num; int scannedItems; while (1) { printf("Enter a number: "); scannedItems = scanf("%lf", &num); while (getchar() != '\n'); if (scannedItems == 1) { return num; } else { printf("Invalid input. Please enter a valid number.\n"); } } } char getOperatorInput() { char op; while (1) { printf("Enter an operator (+, -, *, /): "); scanf(" %c", &op); while (getchar() != '\n'); // Clear input buffer if (op == '+' || op == '-' || op == '*' || op == '/') { return op; } else { printf("Invalid operator. Please enter +, -, *, or /.\n"); } } } double calculate(double num1, double num2, char operator) { switch (operator) { case '+': return num1 + num2; case '-': return num1 - num2; case '*': return num1 * num2; case '/': if (num2 == 0) { printf("Error: Division by zero is not allowed.\n"); return -99999.0; } return num1 / num2; default: printf("Error: Unexpected operator.\n"); return -99999.0; } }
